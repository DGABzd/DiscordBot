const Discord = require("discord.js");

module.exports = {
  name: "configuracion",
  aliases: ["config"],
  category: "Configuracion",
  description: "Ver la configuracion del servidor",
  usage: "",
  run: async (client, message, args, color) => {
    

    const embed = new Discord.MessageEmbed()
      .setTitle("Configuracion del Servidor")
      .setAuthor(message.guild.name, message.guild.iconURL)
      .setColor(color)
      .setImage(message.guild.bannerURL)
      .addField("Prefix", `${client.settings.get(message.guild.id, "prefix")}`)
      .addField("Canal de Registro", client.settings.get(message.guild.id, "logs") ? `<#${client.settings.get(message.guild.id, "logs")}>` : "no Configurado")
      .addField("Rol Moderador", `none`)
      message.channel.send({ embeds: [embed] })
  },
};
